154  private static FovCone ComputeFoVForRange(
155  int observerHeight,
156  FovCone cone,
157  Func<HexCoords,bool> isOnBoard,
158  Func<HexCoords, int> targetHeight,
159  Func<HexCoords, int> terrainHeight,
160  Action<HexCoords> setFieldOfView,
161  FovQueue queue)
162  {
163  Action<FovCone> enqueue = queue.Enqueue;
164 
165  var range = cone.Range;
166  var topVector = cone.VectorTop;
167  var topRiseRun = cone.RiseRun;
168  var bottomVector = cone.VectorBottom;
169 
170  // track the overlap-cone between adjacent hexes as we move down.
171  var overlapVector = cone.VectorTop;
172  var hexX = XFromVector(range, topVector);
173  #if TraceFOV
174  TraceFlag.FieldOfView.Trace(false, "DQ: ({0}) from {1}", cone, hexX);
175  #endif
176 
177  do {
178  while (overlapVector.GT(bottomVector)) {
179  var coordsCurrent = HexCoords.NewCanonCoords(hexX,range);
180  var hexVectorBottom = VectorHexBottom(coordsCurrent);
181  if (isOnBoard(coordsCurrent)) {
182  #region Set current hex parameters
183  var hexVectorTop = VectorHexTop(coordsCurrent);
184  var hexElevation = targetHeight(coordsCurrent);
185  var hexHeight = terrainHeight(coordsCurrent);
186  var hexRiseRun = new RiseRun(hexHeight-observerHeight, range);
187  #endregion
188 
189  #region Check visibility of current hex
190  var riseRun = new RiseRun(hexElevation-observerHeight, GetRange(coordsCurrent));
191  if ( riseRun >= cone.RiseRun
192  && bottomVector.LE(coordsCurrent.Canon) && coordsCurrent.Canon.LE(topVector)
193  ) {
194  setFieldOfView(coordsCurrent);
195  #if TraceFOV
196  TraceFlag.FieldOfView.Trace(false," Set visible: {0} / {1}; {2} >= {3}",
197  MapCoordsDodecant(coordsCurrent), coordsCurrent.ToString(), riseRun, cone.RiseRun);
198  #endif
199  }
200  #endregion
201 
202  #region Check hex transition
203  if (hexRiseRun > topRiseRun) {
204  topVector = LogAndEnqueue(enqueue, range, topVector, hexVectorTop, topRiseRun, 0);
205  topRiseRun = hexRiseRun;
206  } else if (hexRiseRun > cone.RiseRun) {
207  topVector = LogAndEnqueue(enqueue, range, topVector, overlapVector, topRiseRun, 1);
208  topRiseRun = hexRiseRun;
209  } else if (hexRiseRun < cone.RiseRun) {
210  topVector = LogAndEnqueue(enqueue, range, topVector, overlapVector, topRiseRun, 2);
211  topRiseRun = cone.RiseRun;
212  }
213  #endregion
214  }
215 
216  overlapVector = VectorMax(hexVectorBottom, bottomVector);
217  if (hexVectorBottom.GT(bottomVector)) -- hexX;
218  }
219 
220  #region Dequeue next cone
221  if (queue.Count == 0) {
222  topVector = LogAndEnqueue(enqueue, range, topVector, bottomVector, topRiseRun, 3);
223  cone = queue.Dequeue();
224  break;
225  } else {
226  cone = queue.Dequeue();
227  if(cone.Range != range) {
228  topVector = LogAndEnqueue(enqueue, range, topVector, bottomVector, topRiseRun, 4);
229  break;
230  }
231  #if TraceFOV
232  TraceFlag.FieldOfView.Trace(false, "DQ: ({0}) from {1}", cone, hexX);
233  #endif
234  }
235  #endregion
236 
237  #region Check cone transition
238  if (cone.RiseRun > topRiseRun) {
239  topVector = LogAndEnqueue(enqueue, range, topVector, cone.VectorTop, topRiseRun, 5);
240  topRiseRun = cone.RiseRun;
241  } else if (cone.RiseRun < topRiseRun) {
242  topVector = LogAndEnqueue(enqueue, range, topVector, overlapVector, topRiseRun, 6);
243  topRiseRun = cone.RiseRun; // TODO Why is this commented out?
244  }
245  #endregion
246 
247  overlapVector = topVector;
248  bottomVector = cone.VectorBottom;
249  } while(true);
250 
251  // Pick-up any cone portion at bottom of range still unprocessed
252  if (topVector.GT(bottomVector))
253  LogAndEnqueue(enqueue, range, topVector, bottomVector, cone.RiseRun, 7);
254 
255  return cone;
256  }
257  }
