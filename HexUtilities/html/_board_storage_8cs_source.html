152  : base (sizeHexes) {

154  for(var y = 0; y < backingStore.Capacity; y++) {
155  backingStore.Add(new List<List<THex>>((MapSizeHexes.Width+_buffer) / _grouping));
156  for(var x = 0; x < backingStore[y].Capacity; x++) {
157  backingStore[y].Add(new List<THex>(_grouping*_grouping));

161  Parallel.For(0, backingStore.Capacity, y => {
162  var boardRow = backingStore[y];
163  for(var x = 0; x < boardRow.Capacity; x++) {
164  var boardCell = backingStore[y][x];
165  for (var i=0; i<_grouping; i++) {

167  var coords = HexCoords.NewUserCoords(x*_grouping+j,y*_grouping+i);
168  boardCell.Add(IsOnboard(coords) ? initializer(coords) : default(THex));
169  }
