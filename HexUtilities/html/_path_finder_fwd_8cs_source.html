156  var path = item.Value;

178  public void Invoke(DirectedPath path, Hexside hexside) {
179  var here = path.PathStep.Hex;
180  var there = here.Board[here.Coords.GetNeighbour(hexside)];
181  if (there != null && ! _open.ContainsKey(there.Coords)) {
182  var cost = _stepCost(there, hexside.Reversed());
183  if (cost > 0) {
184  var newPath = path.AddStep(there, hexside, cost);
185  var estimate = Estimate(_heuristic, _vectorGoal, _start.Coords, newPath.PathStep.Hex.Coords, newPath.TotalCost);
186  TraceFlags.FindPathEnqueue.Trace(" Enqueue {0}: estimate={1,4}:{2,3}",
187  there.Coords, estimate>>16, estimate & 0xFFFF);
188  _queue.Enqueue(estimate, newPath);
189  }
190  }
