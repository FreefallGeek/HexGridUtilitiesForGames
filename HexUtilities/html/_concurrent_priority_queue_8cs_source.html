284  public void Insert(HexKeyValuePair<TKey,TValue> entry)
285  {
286  // Add the item to the list, making sure to keep track of where it was added.
287  _items.Add(entry);
288  int pos = _items.Count - 1;
289 
290  // If the new item is the only item, we're done.
291  if (pos == 0) return;
292 
293  // Otherwise, perform log(n) operations, walking up the tree, swapping
294  // where necessary based on key values
295  while (pos > 0)
296  {
297  // Get the next position to check
298  int nextPos = (pos-1) / 2;
299 
300  // Extract the entry at the next position
301  var toCheck = _items[nextPos];
302 
303  // Compare that entry to our new one. If our entry has a smaller key, move it up.
304  // Otherwise, we're done.
305  if (entry.Key.CompareTo(toCheck.Key) < 0)
306  {
307  _items[pos] = toCheck;
308  pos = nextPos;
309  }
310  else break;
311  }
312 
313  // Make sure we put this entry back in, just in case
314  _items[pos] = entry;
315  }

326  public HexKeyValuePair<TKey, TValue> Remove()
327  {
328  // Get the first item and save it for later (this is what will be returned).
329  if (_items.Count == 0) throw new InvalidOperationException("The heap is empty.");
330  HexKeyValuePair<TKey, TValue> toReturn = _items[0];
331 
332  // Remove the first item if there will only be 0 or 1 items left after doing so.
333  if (_items.Count <= 2) _items.RemoveAt(0);
334  // A reheapify will be required for the removal
335  else
336  {
337  // Remove the first item and move the last item to the front.
338  _items[0] = _items[_items.Count - 1];
339  _items.RemoveAt(_items.Count - 1);
340 
341  // Start reheapify
342  int current = 0, possibleSwap = 0;
343 
344  // Keep going until the tree is a heap
345  while (true)
346  {
347  // Get the positions of the node's children
348  int leftChildPos = 2 * current + 1;
349  int rightChildPos = leftChildPos + 1;
350 
351  // Should we swap with the left child?
352  if (leftChildPos < _items.Count)
353  {
354  // Get the two entries to compare (node and its left child)
355  var entry1 = _items[current];
356  var entry2 = _items[leftChildPos];
357 
358  // If the child has a lower key than the parent, set that as a possible swap
359  if (entry2.Key.CompareTo(entry1.Key) < 0) possibleSwap = leftChildPos;
360  }
361  else break; // if can't swap this, we're done
362 
363  // Should we swap with the right child? Note that now we check with the possible swap
364  // position (which might be current and might be left child).
365  if (rightChildPos < _items.Count)
366  {
367  // Get the two entries to compare (node and its left child)
368  var entry1 = _items[possibleSwap];
369  var entry2 = _items[rightChildPos];
370 
371  // If the child has a lower key than the parent, set that as a possible swap
372  if (entry2.Key.CompareTo(entry1.Key) < 0) possibleSwap = rightChildPos;
373  }
374 
375  // Now swap current and possible swap if necessary
376  if (current != possibleSwap)
377  {
378  var temp = _items[current];
379  _items[current] = _items[possibleSwap];
380  _items[possibleSwap] = temp;
381  }
382  else break; // if nothing to swap, we're done
383 
384  // Update current to the location of the swap
385  current = possibleSwap;
386  }
387  }
388 
389  // Return the item from the heap
390  return toReturn;
391  }
