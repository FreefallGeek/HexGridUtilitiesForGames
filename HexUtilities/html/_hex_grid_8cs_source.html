100  public virtual HexCoords GetHexCoords(Point point, Size autoScroll) {
101  if( Host == null ) return HexCoords.EmptyCanon;
102 
103  // Adjust for origin not as assumed by GetCoordinate().
104  var grid = new Size((int)(Host.GridSizeF.Width*2F/3F), (int)Host.GridSizeF.Height);
105  var margin = new Size((int)(Host.MapMargin.Width * Host.MapScale),
106  (int)(Host.MapMargin.Height * Host.MapScale));
107  point -= autoScroll + margin + grid;
108 
109  return HexCoords.NewCanonCoords( GetCoordinate(matrixX, point),
110  GetCoordinate(matrixY, point) );
111  }
112 
114  public virtual Point ScrollPositionToCenterOnHex(HexCoords coordsNewCenterHex) {
115  return HexCenterPoint(HexCoords.NewUserCoords(
116  coordsNewCenterHex.User - ( new IntVector2D(Host.VisibleRectangle.Size.User) / 2 )
117  ));
118  }
119 
121  public virtual Point HexCenterPoint(HexCoords coordsNewULHex) {
122  if (coordsNewULHex == null) return new Point();
123  var offset = new Size((int)(Host.GridSizeF.Width*2F/3F), (int)Host.GridSizeF.Height);
124  var margin = Size.Round( Host.MapMargin.Scale(Host.MapScale) );
125  return HexOrigin(coordsNewULHex) + margin + offset;
126  }

150  static int GetCoordinate (Matrix matrix, Point point){
151  var pts = new Point[] {point};
152  matrix.TransformPoints(pts);
153  return (int) Math.Floor( (pts[0].X + pts[0].Y + 2F) / 3F );
154  }

156  Point HexOrigin(HexCoords coords) {
157  return new Point(
158  (int)(Host.GridSizeF.Width * coords.User.X),
159  (int)(Host.GridSizeF.Height * coords.User.Y + Host.GridSizeF.Height/2 * (coords.User.X+1)%2)
160  );
161  }
