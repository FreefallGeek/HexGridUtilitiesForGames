102  public static IPath FindPath(
103  HexCoords start,
104  HexCoords goal,
105  Func<HexCoords, Hexside, int> stepCost,
106  Func<int,int> heuristic,
107  Func<HexCoords,bool> isOnboard
108  ) {
109  if (stepCost==null) throw new ArgumentNullException("stepCost");
110  if (heuristic==null) throw new ArgumentNullException("heuristic");
111  if (isOnboard==null) throw new ArgumentNullException("isOnboard");
112 
113  var vectorGoal = goal.Canon - start.Canon;
114  var closed = new HashSet<HexCoords>();
115  var open = new HashSet<HexCoords>();
116  var queue = DictionaryPriorityQueue<int, Path>.NewQueue();
117  TraceFlags.FindPathDetail.Trace(true, "Find path from {0} to {1}; vectorGoal = {0}",
118  start.Canon, goal.Canon, vectorGoal);
119 
120  queue.Enqueue (0, new Path(start));
121 
122  HexKeyValuePair<int,Path> item;
123  while (queue.TryDequeue(out item)) {
124  var path = item.Value;
125  if( closed.Contains(path.StepCoords) ) continue;
126 
127  open.Add(item.Value.StepCoords);
128 
129  TraceFlags.FindPathDetail.Trace(
130  "Dequeue Path at {0} w/ cost={1,4} at {2}; estimate={3,4}:{4,4}.",
131  item.Value.StepCoords, item.Value.TotalCost, item.Value.HexsideExit,
132  item.Key >> 16, item.Key & 0xFFFF);
133 
134  if(path.StepCoords.Equals(goal)) return path;
135 
136  closed.Add(path.StepCoords);
137 
138  foreach (var neighbour in path.StepCoords.GetNeighbours()
139  .Where(n => isOnboard(n.Coords))
140  ) {
141  var cost = stepCost(path.StepCoords, neighbour.Hexside);
142  if (cost > 0) {
143  var newPath = path.AddStep(neighbour, (ushort)cost);
144  var estimate = Estimate(heuristic,vectorGoal,goal, newPath.StepCoords, newPath.TotalCost);
145  TraceFlags.FindPathDetail.Trace(" Enqueue {0}: {1,4}:{2,3}",
146  neighbour.Coords, estimate>>16, estimate & 0xFFFF);
147  queue.Enqueue(estimate, newPath);
148  }
149  }
150  }
151  return null;
152  }

154  internal static int Estimate(Func<int,int> heuristic, IntVector2D vectorGoal, HexCoords goal,
155  HexCoords hex, int totalCost) {
156  var estimate = (int)heuristic(goal.Range(hex)) + totalCost;
157  var preference = Preference(vectorGoal, goal, hex);
158  return (estimate << 16) + preference;
159  }
160  static int Preference(IntVector2D vectorGoal, HexCoords goal, HexCoords hex) {
161  return (0xFFFF & Math.Abs(vectorGoal ^ (goal.Canon - hex.Canon) ));
162  }
