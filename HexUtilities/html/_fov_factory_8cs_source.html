77  public static IFov GetFieldOfView(IFovBoard<IHex> board, HexCoords origin, FovTargetMode targetMode) {
78  TraceFlags.FieldOfView.Trace("GetFieldOfView");
79  var fov = new ArrayFieldOfView(board);
80  if (board.IsPassable(origin)) {
81  Func<HexCoords,int> target;
82  int observer;
83  switch (targetMode) {
84  case FovTargetMode.EqualHeights:
85  observer = board[origin].ElevationASL + 1;
86  target = coords => board[coords].ElevationASL + 1;
87  break;
88  case FovTargetMode.TargetHeightEqualZero:
89  observer = board[origin].HeightObserver;
90  target = coords => board[coords].ElevationASL;
91  break;
92  default:
93  case FovTargetMode.TargetHeightEqualActual:
94  observer = board[origin].HeightObserver;
95  target = coords => board[coords].HeightTarget;
96  break;
97  }
98  ShadowCasting.ComputeFieldOfView(
99  origin,
100  board.FovRadius,
101  observer,
102  coords => board.IsOnboard(coords),
103  target,
104  coords => board[coords].HeightTerrain,
105  coords => fov[coords] = true
106  );
107  }
108  return fov;
109  }
