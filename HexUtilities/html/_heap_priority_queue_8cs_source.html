145  public MinListHeap(List<HexKeyValuePair<TKey,TValue>> list) {
146  _items = list;
147 
148  // Min-Heapify by sifting-down from last parent in heap.
149  var start = (list.Count - 1) / 2;
150  while (start >= 0) {
151  var root = start;
152  var end = list.Count - 1;
153 
154  while (root*2 + 1 <= end) {
155  var child = root*2 + 1; // (root*2 + 1 points to the left child)
156  var swap = root; // (keeps track of child to swap with)
157 
158  // (check if root is smaller than left child)
159  if (list[swap].CompareTo(list[child]) > 0)
160  swap = child;
161 
162  // (check if right child exists, and if it's bigger than what we're currently swapping with)
163  if (child+1 <= end && list[swap].CompareTo(list[child+1]) > 0 )
164  swap = child + 1;
165 
166  // (check if we need to swap at all)
167  if (swap == root) break;
168 
169  // swap and repeat to continue sifting down the child now
170  var temp = list[root]; list[root] = list[swap]; list[swap] = temp;
171  root = swap;
172  }
173 
174  start = start - 1;
175  }
176  }

201  public HexKeyValuePair<TKey,TValue> ExtractFirst() {

222  void MinHeapifyDownn(int current) {
223  while (true) {
224 
225  // identify smallest of parent and borh children
226  int leftChild = 2 * current + 1, smallest;
227  if (leftChild < _items.Count) smallest = GetSmallest(leftChild,current);
228  else break;
229 
230  int rightChild = leftChild + 1;
231  if (rightChild < _items.Count) smallest = GetSmallest(rightChild,smallest);
232 
233  // if nothing to swap, we're done
234  if (current == smallest) break;
235 
236  // swap smallest value up
237  var temp = _items[current];
238  _items[current] = _items[smallest];
239  _items[smallest] = temp;
240 
241  // follow swapped value down and repeat until the tree is a heap
242  current = smallest;
243  }
244  }
