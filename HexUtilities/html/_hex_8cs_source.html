166  public static IDirectedPath GetDirectedPath(this IHex @this, IHex goal) {
167  if (@this==null) throw new ArgumentNullException("this");
168  if (goal==null) throw new ArgumentNullException("goal");
169  if (! @this.Board.IsPassable(goal.Coords) || ! @this.Board.IsPassable(@this.Coords))
170  return null;
171 
172  return goal.Coords.Range(@this.Coords) > @this.Board.RangeCutoff
173  ? BidirectionalPathfinder.FindDirectedPathFwd(@this, goal, @this.Board)
174  : Pathfinder.FindDirectedPath(@this, goal, @this.Board);
