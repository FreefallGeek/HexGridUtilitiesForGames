89  void FillLandmark(IBoard<IHex> board) {
90  var start = board[Coords];
91  var queue = DictionaryPriorityQueue<int, IHex>.NewQueue();
92  TraceFlags.FindPathDetail.Trace(true, "Find distances from {0}", start.Coords);
93 
94  queue.Enqueue (0, start);
95 
96  HexKeyValuePair<int,IHex> item;
97  while (queue.TryDequeue(out item)) {
98  var here = item.Value;
99  var key = item.Key;
100  if( backingStore[here.Coords] > 0 ) continue;
101 
102  TraceFlags.FindPathDetail.Trace("Dequeue Path at {0} w/ cost={1,4}.", here, key);
103  backingStore[here.Coords] = (short)key;
104 
105  foreach (var there in here.GetAllNeighbours().Where(n => n!=null && n.Hex.IsOnboard())) {
106  var cost = board.DirectedStepCost(here, there.HexsideEntry);
107  if (cost > 0 && backingStore[there.Hex.Coords] == -1) {
108  TraceFlags.FindPathDetail.Trace(" Enqueue {0}: {1,4}", there.Hex.Coords, cost);
109  queue.Enqueue(key + cost, there.Hex);
110  }
111  }
112  }
